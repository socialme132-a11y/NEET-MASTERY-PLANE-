FROM php:8.2-apache

# Install system dependencies
RUN apt-get update && apt-get install -y \
    git \
    curl \
    libpng-dev \
    libonig-dev \
    libxml2-dev \
    zip \
    unzip \
    && docker-php-ext-install mbstring

# Clean up
RUN apt-get clean && rm -rf /var/lib/apt/lists/*

# Copy application files
COPY . /var/www/html/

# Set proper permissions
RUN chown -R www-data:www-data /var/www/html \
    && chmod -R 755 /var/www/html \
    && chmod 664 /var/www/html/users.json \
    && chmod 664 /var/www/html/error.log

# Enable Apache rewrite module
RUN a2enmod rewrite

# Create initial files if they don't exist
RUN if [ ! -f /var/www/html/users.json ]; then echo '{}' > /var/www/html/users.json; fi && \
    if [ ! -f /var/www/html/error.log ]; then touch /var/www/html/error.log; fi && \
    chown www-data:www-data /var/www/html/users.json /var/www/html/error.log && \
    chmod 664 /var/www/html/users.json /var/www/html/error.log

EXPOSE 80
